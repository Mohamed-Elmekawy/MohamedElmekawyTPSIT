    <script>
        // Funzione per stampare gli elementi di un array in div
        function printArrayInDivs(arr) {
            const container = document.getElementById('container');
            
            // Per ogni elemento dell'array, crea un div, aggiungi la classe 'box' e il testo, poi aggiungilo al container
            arr.forEach(element => {
                const div = document.createElement('div');
                div.classList.add('box');
                div.textContent = element;
                container.appendChild(div);
            });
        }

        // Esempio di utilizzo della funzione con un array di numeri e stringhe
        const myArray = [1, 2, 3, 4, 5, "ciao", "test"];
        printArrayInDivs(myArray);
    </script>
